47 void check_frames_are_different(Frame& frm1, Frame& frm2) {
48  const FrameItemType* items1 = frm1.getAs<FrameItemType>();
49  const FrameItemType* items2 = frm2.getAs<FrameItemType>();
50  size_t items_count1 = frm1.size<FrameItemType>();
51  size_t items_count2 = frm2.size<FrameItemType>();
52  if (items_count1 != items_count2) {
53  CHECK(items_count1 != items_count2); // mark success
54  return; // do not test further
55  }
56 
57  float frame_items_count = (float)items_count1; // only cast to float
58  int the_same_x_values = 0;
59  int the_same_y_values = 0;
60  int the_same_z_values = 0;
61  for (int i = 0; i < items_count1; ++i, ++items1, ++items2) {
62  if (items1->x != items2->x) the_same_x_values++;
63  if (items1->y != items2->y) the_same_y_values++;
64  if (items1->z != items2->z) the_same_z_values++;
65  }
66  CHECK(the_same_x_values / frame_items_count < 0.01);
67  CHECK(the_same_y_values / frame_items_count < 0.01);
68  CHECK(the_same_z_values / frame_items_count < 0.01);
69 }

71 template <typename FrameItemType>
72 void check_frame_basics(Frame& frm, size_t items_count)
73 {
74  REQUIRE(items_count > 0);
75 
76  int zeroUid = 0, notZeroX = 0, notZeroY = 0, positiveZ = 0,
77  xyzDiffs = 0; // few trivial data statistics
78  unsigned long int uidSum=0;
79 
80  const FrameItemType* items = frm.getAs<FrameItemType>();
81  for (int i = 0; i < items_count; ++i, ++items) {
82  uidSum+=items->uid;
83  if (items->uid == 0) zeroUid++;
84  if (items->x != 0) notZeroX++;
85  if (items->y != 0) notZeroY++;
86  if (items->z > 0) positiveZ++;
87  if (items->x != items->y || items->y != items->z || items->x != items->z)
88  xyzDiffs++;
89  }
90 
91  float frame_items_count = (float)items_count; // only cast to float
92  CHECK(xyzDiffs / frame_items_count > 0.99);
93  CHECK(notZeroX / frame_items_count > 0.99);
94  CHECK(notZeroY / frame_items_count > 0.99);
95  CHECK(positiveZ / frame_items_count > 0.99);
96  // uids are not sorted, may not be continuous, but should be unique right?
97  // if uids occupy the lowest range <0,items_count) without missing value,
98  // their sum would be ~ 0.5*items_count*items_count => check at least that
99  CHECK(zeroUid <= 1); // max 1 zero uid
100  CHECK(uidSum >= (0.49 * (items_count-1) * items_count));
101 
102 }
