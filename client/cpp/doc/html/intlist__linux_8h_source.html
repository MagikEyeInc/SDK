44  void listener() {
45  // Create unix socket for stopping
46 
47  if ((unix_socket_fd_ = socket(AF_UNIX, SOCK_STREAM, 0)) == -1)
48  throw mke::Error("Cannot open UNIX socket");
49 
50  memset(&un_addr, 0, sizeof(un_addr));
51  un_addr.sun_family = AF_UNIX;
52 
53  *un_addr.sun_path = '\0';
54  int err = -1;
55  for(int i = 0; i < 64; ++i)
56  {
57  snprintf(un_addr.sun_path + 1, sizeof(un_addr.sun_path) - 2, socket_format_ + 1, i);
58 
59  err = bind(unix_socket_fd_, (struct sockaddr*) &un_addr, sizeof(un_addr));
60  if(err != -1)
61  break;
62  }
63 
64  if(err == -1)
65  throw mke::Error("Cannot bind UNIX socket");
66 
67  if (listen(unix_socket_fd_, 1) < 0)
68  throw mke::Error("Cannot listen to UNIX socket");
69 
70  // Create netlink socket for interface notification
71  struct sockaddr_nl nl_addr;
72  int len;
73  char buffer[4096];
74  struct nlmsghdr *nlh;
75 
76  if ((netlink_socket_fd_ = socket(PF_NETLINK, SOCK_RAW, NETLINK_ROUTE)) == -1)
77  throw mke::Error("Cannot open NETLINK_ROUTE socket");
78 
79  memset(&nl_addr, 0, sizeof(nl_addr));
80  nl_addr.nl_family = AF_NETLINK;
81  nl_addr.nl_groups = RTMGRP_IPV4_IFADDR;
82 
83  if (bind(netlink_socket_fd_, (struct sockaddr *) &nl_addr, sizeof(nl_addr)) == -1)
84  throw mke::Error("Cannot bind NETLINK_ROUTE socket");
85 
86  nlh = (struct nlmsghdr *) buffer;
87 
88  // Run select loop
89  fd_set readfds;
90 
91  for (;;)
92  {
93  FD_ZERO(&readfds);
94  FD_SET(unix_socket_fd_, &readfds);
95  FD_SET(netlink_socket_fd_, &readfds);
96 
97  int activity = select(std::max(unix_socket_fd_, netlink_socket_fd_) + 1,
98  &readfds, NULL, NULL, NULL);
99 
100  if ((activity < 0) && (errno != EINTR))
101  throw mke::Error("Select error");
102 
103  if (FD_ISSET(unix_socket_fd_, &readfds))
104  break;
105 
106  if (FD_ISSET(netlink_socket_fd_, &readfds))
107  {
108  if ((len = recv(netlink_socket_fd_, nlh, 4096, 0)) > 0)
109  {
110  while ((NLMSG_OK(nlh, len)) && (nlh->nlmsg_type != NLMSG_DONE))
111  {
112  if (nlh->nlmsg_type == RTM_NEWADDR)
113  {
114  struct ifaddrmsg *ifa = (struct ifaddrmsg *) NLMSG_DATA(nlh);
115  struct rtattr *rth = IFA_RTA(ifa);
116  int rtl = IFA_PAYLOAD(nlh);
117 
118  while (rtl && RTA_OK(rth, rtl))
119  {
120  if (rth->rta_type == IFA_LOCAL)
121  {
122  // we apparently need to wait for some time for the multicast group to appear...
123  std::this_thread::sleep_for(std::chrono::milliseconds(500));
124 
125  if (interface_update_callback_)
126  interface_update_callback_();
127  }
128 
129  rth = RTA_NEXT(rth, rtl);
130  }
131  }
132  else if (nlh->nlmsg_type == RTM_DELADDR)
133  {
134  struct ifaddrmsg *ifa = (struct ifaddrmsg *) NLMSG_DATA(nlh);
135  struct rtattr *rth = IFA_RTA(ifa);
136  int rtl = IFA_PAYLOAD(nlh);
137 
138  while (rtl && RTA_OK(rth, rtl))
139  {
140  if (rth->rta_type == IFA_LOCAL)
141  {
142  if (interface_update_callback_)
143  interface_update_callback_();
144  }
145  rth = RTA_NEXT(rth, rtl);
146  }
147  }
148 
149  nlh = NLMSG_NEXT(nlh, len);
150  }
151  }
152  }

185  // Signal the unix socket to break out of the listener thread
186  int socket_fd;
187 
188  if ((socket_fd = socket(AF_UNIX, SOCK_STREAM, 0)) == -1)
189  throw mke::Error("Cannot open UNIX socket");

210  void update() {
211  Interface *interface_ptr = nullptr;
212 
213  // Iterate through the list and delete removed interfaces
214  deleteRemovedInterfaces();
215 
216  // List available network interfaces
217  struct ifaddrs *ifaddr, *ifa;
218  int n;
219 
220  if (getifaddrs(&ifaddr) == -1)
221  throw mke::Error("Cannot call getifaddrs");
222 
223  for (n = 0, ifa = ifaddr; ifa != NULL; ifa = ifa->ifa_next, n++)
224  {
225  if (ifa->ifa_addr == NULL)
226  continue;
227 
228  try
229  {
230  if (ifa->ifa_addr->sa_family == AF_INET)
231  {
232  struct sockaddr_in *sockaddr = (struct sockaddr_in *) (ifa->ifa_addr);
233  uint32_t s_addr = sockaddr->sin_addr.s_addr;
234  auto interface = interfaces_.find(s_addr);
235 
236  if (interface != interfaces_.end())
237  {
238  Interface *interface_ptr = interface->second;
239  interface_ptr->setStatus(Interface::UPDATED);
240  }
241  else
242  {
243  char interface_ipv4[INET_ADDRSTRLEN + 1];
244  inet_ntop(AF_INET, &(sockaddr->sin_addr), interface_ipv4, INET_ADDRSTRLEN);
245  interface_ptr = new Interface(interface_ipv4, s_addr, multicast_ipv4_, multicast_pv4_, io_service_);
246  interface_ptr->start();
247  interfaces_[s_addr] = interface_ptr;
248  }
249  }
250  }
251  catch(std::exception & e)
252  {
253  std::cerr << "SSDP error (ignored): Unable to add interface into multicast group. Note: " << e.what() << std::endl;
254  }
255  }
256 
257  freeifaddrs(ifaddr);
258 
259  // Iterate through once again to stop the removed interfaces
260  stopRemovedInterfaces();
261  }
