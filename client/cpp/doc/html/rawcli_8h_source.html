141 template<class SyncType>
142 SyncType RawClient::syncCall(std::function<void(std::promise<SyncType> & p, Client::Handle & op_handle)> ftor,
143  unsigned timeout_ms)
144 {
145  std::promise<SyncType> p;
146  std::future<SyncType> ret = p.get_future();
147 
148  Client::Handle op_handle;
149  ftor(p, op_handle);
150 
151  if(timeout_ms != TIMEOUT_INFINITE &&
152  ret.wait_for(std::chrono::milliseconds(timeout_ms)) == std::future_status::timeout)
153  {
154  if (cancelRequest(op_handle))
155  throw mke::Error("Request timed out");
156  }
157 
158  return ret.get();
159 }

163 template<class SyncType>
164 ErrorCallback RawClient::syncErrorCallback(std::promise<SyncType> & p)
165 {
166  return [&p](const mke::Error & e)
167  {
168  try {
169  if (dynamic_cast<const mke::cli::IOError*>(&e)) {
170  throw dynamic_cast<const mke::cli::IOError&>(e);
171  } else if (dynamic_cast<const mke::cli::BadReplyError*>(&e)) {
172  throw dynamic_cast<const mke::cli::BadReplyError&>(e);
173  } else if (dynamic_cast<const mke::cli::ServerFatalError*>(&e)) {
174  throw dynamic_cast<const mke::cli::ServerFatalError&>(e);
175  } else {
176  throw dynamic_cast<const mke::Error&>(e);
177  }
178  } catch (const mke::cli::IOError&) {
179  p.set_exception(std::current_exception());
180  } catch (const mke::cli::BadReplyError&) {
181  p.set_exception(std::current_exception());
182  } catch (const mke::cli::ServerFatalError&) {
183  p.set_exception(std::current_exception());
184  } catch (const mke::Error&) {
185  p.set_exception(std::current_exception());
186  } catch (...) {
187  p.set_exception(std::current_exception());
188  }
189  };
