113 #if (USE_OPENCV)
120  template <typename FrameItemType>
121  void getOpenCV(cv::Mat& mat) const {
122  size_t pcSize = size<FrameItemType>();
123  const FrameItemType* frame = reinterpret_cast<const FrameItemType*>(data);
124  mat.resize(pcSize);
125  if (mat.type()==CV_32FC3) { // 3x float
126  cv::Point3f* data = mat.ptr<cv::Point3f>();
127  for(size_t i = 0; i < pcSize; i++) {
128  data[i] = cv::Point3f(frame[i].x,frame[i].y,frame[i].z);
129  }
130  }
131  else if (mat.type()==CV_32SC3) { // 3x signed 32bit integer
132  cv::Point3i* data = mat.ptr<cv::Point3i>();
133  for(size_t i = 0; i < pcSize; i++) {
134  data[i] = cv::Point3i(frame[i].x,frame[i].y,frame[i].z);
135  }
136  }
137  else if (mat.type()==CV_16SC3) { // 3x signed 16bit integer
138  int16_t* data = mat.ptr<int16_t>();
139  for(size_t i = 0; i < pcSize; i++) {
140  size_t dest=i*3;
141  data[dest] = frame[i].x;
142  data[dest+1] = frame[i].y;
143  data[dest+2] = frame[i].z;
144  }
145  }
146  else {
147  throw mke::Error("Unsupported cv::Mat type");
148  }
149  }
